口诀

外层控制列，内层控制行，外层乘内层，外层再换行，while计数器，末尾要自加

两层for嵌套

for i in range(1,10):
    for j in range(1,i+1):
        print(f'{j}*{i}={j*i}',end=" ")
    print()
<?php
for ($i = 1; $i <= 9; $i++) {
    for ($j = 1; $j <= $i; $j++) {
        $k = $i * $j;
        echo "$j*$i=$k ";
    }
    echo "\n";
}
?>
#include<stdio.h>
int main(){
    for (int i = 1; i <= 9; i++){
        for (int j = 1; j <= i; j++) {
            printf("%d*%d=%d ", j, i, i * j);
        }
        printf("\n");
    }
        return 0;
}

两层while嵌套

i=1
while(i <= 9):
    j=1
    while(j<=i):
        print(f'{j}*{i}={j*i}',end=" ")
        j+=1
    print()
    i+=1
<?php
$i = 1;
while ($i <= 9) {
    $j = 1;
    while ($j <= $i) {
        $k = $i * $j;
        echo "$j*$i=$k ";
        $j++;
    }
    $i++;
    echo "\n";
}
?>
#include <stdio.h>
int main() 
{
    int i=1;
    while(i<=9){
        int j=1;
        while(j<=i){
            printf("%d*%d=%d ",j,i,i*j);
            j++;
        }
        printf("\n");
        i++;
    }
    return 0;
}
